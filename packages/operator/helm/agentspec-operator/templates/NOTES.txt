 k9s drill-down plugins (optional)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
Add to ~/.config/k9s/plugins.yaml:

plugins:
  agentspec-health:
    shortCut: Shift-H
    description: Full health report (sidecar)
    scopes: [agentobservations]
    command: bash
    background: false
    args:
      - -c
      - |
        SVC=$(kubectl get ao $NAME -n $NAMESPACE -o jsonpath='{.spec.sidecarServiceName}')
        PORT=$(kubectl get ao $NAME -n $NAMESPACE -o jsonpath='{.spec.sidecarPort}')
        kubectl run agentspec-probe-$RANDOM --rm -i --restart=Never \
          --image=curlimages/curl -- \
          curl -s http://$SVC.$NAMESPACE.svc.cluster.local:$PORT/health/ready \
          | python3 -m json.tool

  agentspec-gap:
    shortCut: Shift-G
    description: Gap analysis (sidecar)
    scopes: [agentobservations]
    command: bash
    background: false
    args:
      - -c
      - |
        SVC=$(kubectl get ao $NAME -n $NAMESPACE -o jsonpath='{.spec.sidecarServiceName}')
        PORT=$(kubectl get ao $NAME -n $NAMESPACE -o jsonpath='{.spec.sidecarPort}')
        kubectl run agentspec-gap-$RANDOM --rm -i --restart=Never \
          --image=curlimages/curl -- \
          curl -s http://$SVC.$NAMESPACE.svc.cluster.local:$PORT/gap \
          | python3 -m json.tool
